import os
import speech_recognition as sr
import pyttsx3
from langchain.chat_models import ChatOpenAI
from langchain.schema import HumanMessage
from text2emotion import get_emotion

# Load your API key from Replit Secrets (named OPENROUTERAI_API)
api_key = os.environ.get("OPENROUTERAI_API")

# Initialize the LLM using OpenRouter with GPT-4
llm = ChatOpenAI(
    openai_api_base="https://openrouter.ai/api/v1",
    openai_api_key=api_key,
    model="openrouter/openai/gpt-4",  # Change this if needed
    temperature=0.7
)

# Initialize the text-to-speech engine
engine = pyttsx3.init()

def emotional_response(text):
    """
    Uses text2emotion to detect the dominant emotion from the given text and returns a supportive prefix.
    """
    emotions = get_emotion(text)
    # Determine the dominant emotion
    dominant = max(emotions, key=emotions.get)
    prefix = ""
    if dominant == "Happy":
        prefix = "I'm so glad to hear that. "
    elif dominant == "Sad":
        prefix = "I'm sorry you're feeling down. "
    elif dominant == "Angry":
        prefix = "It's okay to feel angry sometimes. "
    elif dominant == "Fear":
        prefix = "I understand your fears. "
    elif dominant == "Surprise":
        prefix = "That sounds unexpected. "
    return prefix

# Set up the SpeechRecognition recognizer
recognizer = sr.Recognizer()

print("Welcome to SOL Space with Voice & Emotion.\nSpeak now (say 'exit' to quit).")

while True:
    with sr.Microphone() as source:
        print("Listening...")
        audio_data = recognizer.listen(source)
        try:
            user_text = recognizer.recognize_google(audio_data)
            print("You said:", user_text)
        except Exception as e:
            print("Sorry, I couldn't understand that. Please try again.", e)
            continue

    if user_text.lower() in ["exit", "quit"]:
        farewell = "Take care. I’m always here when you need me."
        print("Sol:", farewell)
        engine.say(farewell)
        engine.runAndWait()
        break

    # Get a supportive tone based on your input's emotion
    emo_prefix = emotional_response(user_text)

    # Query the LLM with your input
    try:
        response = llm([HumanMessage(content=user_text)])
        assistant_reply = response.content
        # Prepend the emotion-based prefix to the reply
        full_reply = emo_prefix + assistant_reply
        print("Sol:", full_reply)
        engine.say(full_reply)
        engine.runAndWait()
    except Exception as e:
        error_msg = "I'm having trouble contacting the assistant right now."
        print("Error:", error_msg, e)
        engine.say(error_msg)
        engine.runAndWait()
        continue
